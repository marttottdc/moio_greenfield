// Make global functions available for onclick handlers
window.generateToken = function(fieldId) {
    const array = new Uint8Array(32);
    crypto.getRandomValues(array);
    const token = Array.from(array, byte => byte.toString(16).padStart(2, '0')).join('');
    const field = document.getElementById(fieldId);
    if (field) field.value = token;
};

window.copyValueToClipboard = function(fieldId, btn) {
    const input = document.getElementById(fieldId);
    if (!input) return;
    
    navigator.clipboard.writeText(input.value)
        .then(() => {
            const originalText = btn.innerHTML;
            btn.innerHTML = '<i class="bi bi-check"></i> Copied!';
            setTimeout(() => btn.innerHTML = originalText, 1500);
        })
        .catch(err => {
            console.error('Copy failed:', err);
            btn.textContent = 'Error';
        });
};

// Define the toggle function that will be called
function initWebhookAuthToggle() {
    const authTypeSelect = document.getElementById('auth_type');
    if (!authTypeSelect) {
        console.warn('auth_type select not found');
        return;
    }
    
    function toggleAuthFields() {
        const authType = authTypeSelect.value;
        const authGroups = document.querySelectorAll('.auth-config-group');

        // Hide all auth config groups
        authGroups.forEach(group => {
            group.style.display = 'none';
        });

        // Show the relevant group
        if (authType && authType !== 'none') {
            const targetGroup = document.getElementById(`${authType}-fields`);
            if (targetGroup) {
                targetGroup.style.display = 'block';
            }
        }
    }
    
    // Set initial state
    toggleAuthFields();
    
    // Attach change event (remove old listeners first)
    authTypeSelect.removeEventListener('change', toggleAuthFields);
    authTypeSelect.addEventListener('change', toggleAuthFields);
}

// Execute when script loads (for normal page loads)
if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', initWebhookAuthToggle);
} else {
    // DOM already loaded, execute immediately
    initWebhookAuthToggle();
}

// Also execute when loaded via HTMX
document.body.addEventListener('htmx:afterSwap', function(event) {
    // Check if this swap contains our webhook form
    if (event.target.querySelector('#auth_type') || event.target.id === 'webhook-modal-content' || event.target.id === 'webhook-builder-modal-content') {
        setTimeout(initWebhookAuthToggle, 0);
    }
});

// Handle form submission success
document.body.addEventListener('htmx:afterRequest', function(event) {
    const form = event.target;
    if (form.matches('form[hx-post*="webhook_config"]') && event.detail.successful && event.detail.xhr.status === 204) {
        // Close the appropriate modal
        const crm_modal = document.getElementById('webhookModal');
        const builder_modal = document.getElementById('webhookBuilderModal');
        
        if (crm_modal) {
            const modal = bootstrap.Modal.getInstance(crm_modal);
            if (modal) modal.hide();
        }
        if (builder_modal) {
            const modal = bootstrap.Modal.getInstance(builder_modal);
            if (modal) modal.hide();
        }
        
        // Trigger a page refresh or update the webhook list
        if (!window.__FLOW_BUILDER_ACTIVE__) {
            setTimeout(() => location.reload(), 100);
        }
    }
});
</script>
